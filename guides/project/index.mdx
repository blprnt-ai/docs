<Accordion title="Creating a new project">
  <Steps>
    <Step title="Open the blprnt desktop app">
      Click on the 'Create New Project' button in the sidebar.
      <Frame>
        <img src="./new-project-sidebar.png" alt="New Project Sidebar" />
      </Frame>
    </Step>
    <Step title="Enter the project name">
      Enter the project name and click on the 'Next' button.
      <Frame>
        <img src="./project-details.png" alt="Project Details" />
      </Frame>
    </Step>
    <Step title="Add the project folder">
      Add the project folder.

      <Frame>
        <img src="./new-project-folder.png" alt="New Project Folder" />
      </Frame>

      <Note>
        The project folder is the sandbox the agent can safely read from and write to.
      </Note>
    </Step>
    <Step title="Agent Primer (Optional)">
      Add an Agent Primer to set default context for every new session.
      <Frame>
        <img src="./agent-primer.png" alt="Agent Primer" />
      </Frame>
    </Step>
  </Steps>
</Accordion>

<Accordion title="Managing projects">
  <Steps>
    <Step title="Listing projects">
      Click on the 'Edit' button in the sidebar next to the project you want to edit.
      <Frame>
        <img src="./projects-sidebar.png" alt="Projects Sidebar" />
      </Frame>
    </Step>
    <Step title="Edit your project">
      Edit the project name, folder, or agent primer.
      ---
      <Badge color="blue">Info</Badge> Settings are saved automatically.

      <Frame>
        <img src="./manage-project.png" alt="Project Details" />
      </Frame>


      <Warning>
        Changing the project folder can have unexpected side effects for previous sessions. Please be careful.
      </Warning>
    </Step>
    <Step title="Deleting your project" icon="trash">
      You can also delete the project by clicking on the 'Delete' button.

      <Danger>
        This will delete all the project's sessions and memories.
      </Danger>

      <Frame>
        <img src="./delete-project.png" alt="Delete Project" />
      </Frame>
    </Step>
  </Steps>
</Accordion>
